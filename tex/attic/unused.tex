\section{Fa\`a di Bruno's formula}

\begin{thm}
\thmtitle{Fa\`a di Bruno's formula}
\begin{align}
  \pd{^n}{x_1\cd \pt x_n}
  f(g(\bm{x}))
=
  \sum_{k=1}^n
  \sum_{(\bm{x}_1,\ld,\bm{x}_k)}^{\mc{P}_k(\bm{x})}
  f\ord{k}(g(\bm{x}))
  \prod_{i=1}^k
  \pd{
    ^{|\bm{x}_i|}
    g(\bm{x})
  }{
    x_{i,1}
  \cd
    \pt
    x_{i,|\bm{x}_i|}
  }
\end{align}
\end{thm}
